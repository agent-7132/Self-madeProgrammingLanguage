// 张量运算SIMD优化示例（AVX-512）  
void tensor_multiply(float* A, float* B, float* C, int size) {  
  pragma omp parallel for  
  for (int i = 0; i < size; i += 16) {  
    __m512 vecA = _mm512_load_ps(&Ai);  
    __m512 vecB = _mm512_load_ps(&Bi);  
    __m512 vecC = _mm512_mul_ps(vecA, vecB);  
    _mm512_store_ps(&Ci, vecC);  
  }
